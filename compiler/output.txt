letter = a-z | A-Z ---> def
digit = 0 - 9 ---> def
id: letter (letter|digit)* ---> exp
digits = digit+ ---> def
{boolean int float} ---> key

relop: \=\= | !\= | > | >\= | < | <\= ---> exp
{ if else while } ---> key
[; , \( \) { } =] ---> op
addop:\+ | - ---> exp
mulop:\* | / ---> exp
